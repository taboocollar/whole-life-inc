class DecisionTreeNavigator:
    def __init__(self, nocturne_config):
        self.trees = nocturne_config['decision_trees']
        
    def navigate(self, tree_name, user_state):
        """Navigate decision tree based on user state"""
        tree = self.trees[tree_name]
        current_node_id = tree['entry_point']
        path = [current_node_id]
        
        while current_node_id:
            node = tree['nodes'][current_node_id]
            
            if node['type'] == 'evaluation':
                current_node_id = self._evaluate_branches(node, user_state)
            elif node['type'] == 'response_mode':
                return {
                    'mode': current_node_id,
                    'characteristics': node.get('characteristics', []),
                    'path': path
                }
            elif node['type'] == 'interaction':
                # Return for user interaction
                return {
                    'mode': current_node_id,
                    'actions': node.get('actions', []),
                    'path': path
                }
            
            if current_node_id:
                path.append(current_node_id)
        
        return {'mode': 'default', 'path': path}
    
    def _evaluate_branches(self, node, user_state):
        """Evaluate which branch to take"""
        # Check user state against branch conditions
        # Simplified logic - full implementation would be more sophisticated
        
        familiarity = user_state.get('familiarity', 'new')
        emotion = user_state.get('emotional_state', 'neutral')
        
        if emotion == 'distressed':
            return 'crisis_support'
        elif familiarity == 'new' and emotion == 'neutral':
            return 'gentle_introduction'
        elif familiarity == 'established' and 'curious' in user_state.get('traits', []):
            return 'deep_engagement'
        
        # Default
        return list(node['branches'].values())[0]
